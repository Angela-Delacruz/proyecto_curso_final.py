# Funcin para obtener el GDP de cualquier pais
def gdp_country(dataframe, country_name):
    # Filtrar la informacin correspondiente al pas
    country_data = dataframe[dataframe['Country Name'] == country_name]

    # Eliminar las columnas especificadas
    country_data = country_data.drop(['Country Name', 'Country Code', 'Series Code'], axis=1)

    # Resetear el ndice
    country_data = country_data.reset_index(drop=True)

    # Transponer el DataFrame
    country_data_transposed = country_data.T

    # Obtener los valores de la fila con ndice 3 y convertirlos a una lista
    gdp_values = country_data_transposed[3].tolist()

    # Eliminar el primer valor (si es necesario)
    gdp_values.pop(0)

    return gdp_values
